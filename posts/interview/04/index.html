func getRank(data []int) (int, int, int, []int) {
	m := make([]int, 15)
	for _, d := range data {
		m[d]++
		if d == 1 {
			m[14]++
		}
	}

	for i := 14; i >= 5; i-- {
		t := 1
		for j := 0; j < 5; j++ {
			t *= m[i-j]
		}
		if t == 1 {
			return i, 0, 0, nil
		}
	}

	cnt := 0
	max := 0
	var maxID []int
	for i := 13; i >= 1; i-- {
		if m[i] > 0 {
			cnt++
			if m[i] > max {
				max = m[i]
				maxID = []int{i}
			} else if m[i] == max {
				maxID = append(maxID, i)
			}
		}
	}
	return 0, cnt, max, maxID
}
