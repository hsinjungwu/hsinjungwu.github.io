package main

import (
	"fmt"
	"time"
)

func main() {
	t := time.Now()
	data := 998
	ans := getAns(data)
	elapsed := time.Since(t)
	fmt.Printf("Elapsed time: %v, data = %d, ans = %d", elapsed, data, ans)
}

func getAns(data int) int {
	r := data % 100
	q := data / 100

	step := 0
	if r == 0 {
		b0 := 100
		for q%10 == 0 {
			b0 *= 10
			q /= 10
		}
		s0 := q % 10
		step = (b0-1)*(10-s0)/9 + 1
		q /= 10
		for q%10 == 9 {
			step += s0
			s0 *= 10
			q /= 10
		}
	} else if r > 90 {
		b9X := 10
		s9X := r % 10
		for q%10 == 9 {
			b9X *= 10
			q /= 10
		}
		step = (b9X-1)*s9X/9 + 1
	} else if r%10 == 0 {
		sX0 := r / 10
		step = 11 - sX0

		for q%10 == 9 {
			step += sX0
			sX0 *= 10
			q /= 10
		}
	} else {
		step = 1
	}

	return data + step*9
}
